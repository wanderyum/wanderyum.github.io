数学

字符串

String s = "01234567890";
s += "123";

String s1 = in.nextLine();  // 读入一行
String s2 = in.next();      // 读入一个单词，以空格(广义)结束

/* 比较 */
s1==s2;         // 比较地址
s1.equals(s2);  // 比较内容
s1.compareTo(s2);   // 比大小: s1大->正数, 相等->0, s1小->负数

s1.length();        // 获取长度
s1.charAt(4);       // 获取指定位置字符
s1.substring(1);    // 从位置1到末尾的子串
s1.substring(2,4);  // 子串, 含2不含4

/* 索引 */
s.indexOf("2");         // 找字符2, 返回位置(整数), 不存在返回-1
s.indexOf("2", 3);      // 找字符2, 从位置3开始找(含位置3), 返回位置(整数)
s.lastIndexOf("2");     // 类似, 从右往左
s.lastIndexOf("2", 3);

s.startsWith("234");
s.endsWith("0123");
s.trim();               // 去掉首尾广义空格
s = s.replace("0", "bbc");
s = s.toUpperCase();
s = s.toLowerCase();

函数

    return (int)(Math.random()*n);
}
/* 调用 */
public static void main(String[] args){
    int num = 50;
    System.out.println(myRandom(num));
}
